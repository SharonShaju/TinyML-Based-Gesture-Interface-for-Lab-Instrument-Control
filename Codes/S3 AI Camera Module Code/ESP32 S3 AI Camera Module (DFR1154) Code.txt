#undef ei_printf
#endif

#include <S7_GCI_Project_inferencing.h>
#include "edge-impulse-sdk/dsp/image/image.hpp"
#include "esp_camera.h"

#include <esp_now.h>
#include <WiFi.h>

// ---------- CAMERA CONFIGURATION FOR DFR1154 (OV3660) ----------
#define PWDN_GPIO_NUM    -1
#define RESET_GPIO_NUM   -1
#define XCLK_GPIO_NUM    5
#define SIOD_GPIO_NUM    8
#define SIOC_GPIO_NUM    9

#define Y9_GPIO_NUM      4
#define Y8_GPIO_NUM      6
#define Y7_GPIO_NUM      7
#define Y6_GPIO_NUM      14
#define Y5_GPIO_NUM      17
#define Y4_GPIO_NUM      21
#define Y3_GPIO_NUM      18
#define Y2_GPIO_NUM      16
#define VSYNC_GPIO_NUM   1
#define HREF_GPIO_NUM    2
#define PCLK_GPIO_NUM    15

#define LED_PIN 3   // Onboard LED
#define IR_LED 47   // Night vision IR LED

// Raw capture size (camera native)
#define EI_CAMERA_RAW_FRAME_BUFFER_COLS  320
#define EI_CAMERA_RAW_FRAME_BUFFER_ROWS  240
#define EI_CAMERA_FRAME_BYTE_SIZE        3

static bool debug_nn = false;
static bool is_initialised = false;
static uint8_t *snapshot_buf = nullptr; // allocated in PSRAM

static camera_config_t camera_config = {
    .pin_pwdn = PWDN_GPIO_NUM,
    .pin_reset = RESET_GPIO_NUM,
    .pin_xclk = XCLK_GPIO_NUM,
    .pin_sscb_sda = SIOD_GPIO_NUM,
    .pin_sscb_scl = SIOC_GPIO_NUM,
    .pin_d7 = Y9_GPIO_NUM,
    .pin_d6 = Y8_GPIO_NUM,
    .pin_d5 = Y7_GPIO_NUM,
    .pin_d4 = Y6_GPIO_NUM,
    .pin_d3 = Y5_GPIO_NUM,
    .pin_d2 = Y4_GPIO_NUM,
    .pin_d1 = Y3_GPIO_NUM,
    .pin_d0 = Y2_GPIO_NUM,
    .pin_vsync = VSYNC_GPIO_NUM,
    .pin_href = HREF_GPIO_NUM,
    .pin_pclk = PCLK_GPIO_NUM,
    .xclk_freq_hz = 20000000,
    .ledc_timer = LEDC_TIMER_0,
    .ledc_channel = LEDC_CHANNEL_0,
    .pixel_format = PIXFORMAT_JPEG,
    .frame_size = FRAMESIZE_QVGA,
    .jpeg_quality = 8,
    .fb_count = 1,
    .fb_location = CAMERA_FB_IN_PSRAM,
    .grab_mode = CAMERA_GRAB_WHEN_EMPTY,
};

// ---------- ESP-NOW Settings ----------
uint8_t receiverMAC[] = {0x84, 0x1F, 0xE8, 0x68, 0x60, 0x48}; // Wroom32 MAC

void sendGesture(const char* gesture) {
    esp_err_t result = esp_now_send(receiverMAC, (uint8_t *)gesture, strlen(gesture));
    if(result == ESP_OK){
        Serial.println("Gesture sent via ESP-NOW");
    } else {
        Serial.println("Error sending gesture");
    }
}

void initESPNowSender() {
    WiFi.mode(WIFI_STA);
    if (esp_now_init() != ESP_OK) {
        Serial.println("Error initializing ESP-NOW");
        return;
    }

    esp_now_peer_info_t peerInfo = {};
    memcpy(peerInfo.peer_addr, receiverMAC, 6);
    peerInfo.channel = 0;
    peerInfo.encrypt = false;

    if (esp_now_add_peer(&peerInfo) != ESP_OK){
        Serial.println("Failed to add peer");
    } else {
        Serial.println("ESP-NOW peer added");
    }
}

// ---------- Setup ----------
void setup() {
    Serial.begin(460800);

    pinMode(LED_PIN, OUTPUT);
    pinMode(IR_LED, OUTPUT);
    digitalWrite(IR_LED, HIGH);

    Serial.println("Edge Impulse Gesture Detection (DFR1154 / OV3660) @460800 baud");

    if (!ei_camera_init()) {
        Serial.println("Failed to initialize Camera!");
        while (1) { delay(1000); }
    }

    Serial.println("Camera initialized. Starting inference loop...");

    initESPNowSender(); // Initialize ESP-NOW sender

    delay(3000);
}

// ---------- Loop ----------
void loop() {
    if (ei_sleep(5) != EI_IMPULSE_OK) return;

    // allocate snapshot buffer once
    if (snapshot_buf == nullptr) {
        size_t buf_sz = EI_CAMERA_RAW_FRAME_BUFFER_COLS * EI_CAMERA_RAW_FRAME_BUFFER_ROWS * EI_CAMERA_FRAME_BYTE_SIZE;
        snapshot_buf = (uint8_t*)ps_malloc(buf_sz);
        if (snapshot_buf == nullptr) {
            Serial.println("ERR: Failed to allocate snapshot_buf in PSRAM");
            delay(500);
            return;
        }
    }

    ei::signal_t signal;
    signal.total_length = EI_CLASSIFIER_INPUT_WIDTH * EI_CLASSIFIER_INPUT_HEIGHT;
    signal.get_data = &ei_camera_get_data;

    digitalWrite(LED_PIN, HIGH);
    bool ok = ei_camera_capture(EI_CLASSIFIER_INPUT_WIDTH, EI_CLASSIFIER_INPUT_HEIGHT, snapshot_buf);
    digitalWrite(LED_PIN, LOW);

    if (!ok) {
        delay(2000);
        return;
    }

    ei_impulse_result_t result = { 0 };
    EI_IMPULSE_ERROR err = run_classifier(&signal, &result, debug_nn);
    if (err != EI_IMPULSE_OK) {
        delay(2000);
        return;
    }

    int max_idx = 0;
    float max_val = result.classification[0].value;
    for (int i = 1; i < EI_CLASSIFIER_LABEL_COUNT; i++) {
        if (result.classification[i].value > max_val) {
            max_val = result.classification[i].value;
            max_idx = i;
        }
    }

    char gestureStr[64];
    snprintf(gestureStr, sizeof(gestureStr), "Detected gesture: %s (%.2f%%)",
             ei_classifier_inferencing_categories[max_idx],
             max_val * 100.0f);

    Serial.println(gestureStr);
    sendGesture(gestureStr); // send via ESP-NOW

    delay(3000); // 3-second delay between runs
}

// ---------- Camera init ----------
bool ei_camera_init(void) {
    if (is_initialised) return true;

    esp_err_t err = esp_camera_init(&camera_config);
    if (err != ESP_OK) {
        Serial.printf("Camera init failed with error 0x%x\n", err);
        return false;
    }

    sensor_t *s = esp_camera_sensor_get();
    if (s->id.PID == OV3660_PID) {
        s->set_vflip(s, 1);
        s->set_brightness(s, 1);
        s->set_saturation(s, 0);
    }

    is_initialised = true;
    return true;
}

// ---------- Deinit ----------
void ei_camera_deinit(void) {
    if (!is_initialised) return;
    esp_camera_deinit();
    is_initialised = false;
}

// ---------- Capture & convert ----------
bool ei_camera_capture(uint32_t img_width, uint32_t img_height, uint8_t *out_buf) {
    if (!is_initialised) return false;

    camera_fb_t *fb = esp_camera_fb_get();
    if (!fb) return false;

    bool converted = fmt2rgb888(fb->buf, fb->len, PIXFORMAT_JPEG, snapshot_buf);
    esp_camera_fb_return(fb);
    if (!converted) return false;

    if (img_width != EI_CAMERA_RAW_FRAME_BUFFER_COLS || img_height != EI_CAMERA_RAW_FRAME_BUFFER_ROWS) {
        int res = ei::image::processing::crop_and_interpolate_image(
            snapshot_buf,
            EI_CAMERA_RAW_FRAME_BUFFER_COLS,
            EI_CAMERA_RAW_FRAME_BUFFER_ROWS,
            out_buf,
            img_width,
            img_height,
            3);
        if (res != 0) {
            return false;
        }
    } else {
        memcpy(out_buf, snapshot_buf, img_width * img_height * 3);
    }

    return true;
}

// ---------- Data callback ----------
static int ei_camera_get_data(size_t offset, size_t length, float *out_ptr) {
    size_t pixel_ix = offset * 3;
    size_t out_ix = 0;
    while (length--) {
        out_ptr[out_ix++] =
            (snapshot_buf[pixel_ix + 2] << 16) +
            (snapshot_buf[pixel_ix + 1] << 8) +
             snapshot_buf[pixel_ix + 0];
        pixel_ix += 3;
    }
    return 0;
}
